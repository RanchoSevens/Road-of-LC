### 417. Pacific Atlantic Water Flow

```java
/ * 20220404 */

/*
 * Input: heights = [[1,2,2,3,5],[3,2,3,4,4],[2,4,5,3,1],[6,7,1,4,5],[5,1,1,2,4]]
 * Output: [[0,4],[1,3],[1,4],[2,2],[3,0],[3,1],[4,0]]
 * 思路：问题分拆成两个子问题 -> 分别求能流向太平洋和大西洋的结点
 *      两个子问题思路相同，根据相邻结点的流向关系
 *      （左/上能流向 + 本结点地势 >= 那个结点，则可以，否则不行)
 */

/* 栈 */
class Solution {
    static int[] dir = {-1, 0, 1, 0, -1};

    public static List<List<Integer>> pacificAtlantic(int[][] heights) {
        int m = heights.length, n = (m == 0 ? m : heights[0].length), x, y;
        int[][] accessPac = new int[m][n];
        int[][] accessAtl = new int[m][n];
        /* 初始赋值，第一列和第一行，均为1 */
        for (int i = 0; i < m; i++) {
            accessPac[i][0] = 1;
        }
        for (int j = 0; j < n; j++) {
            accessPac[0][j] = 1;
        }

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (accessPac[i][j] == 1) {
                    Stack<int[]> st = new Stack<>();
                    st.push(new int[]{i, j});
                    while (!st.empty()) {
                        int[] point = st.pop();
                        for (int k = 0; k < 4; k++) {
                            x = point[0] + dir[k];
                            y = point[1] + dir[k + 1];
                            if (x < 0 || x >= m || y < 0 || y >= n) continue;
                            if (accessPac[x][y] == 0 &&
                                    heights[x][y] >= heights[point[0]][point[1]]) {
                                accessPac[x][y] = 1;
                                st.push(new int[]{x, y});
                            }
                        }
                    }
                }
            }
        }

        /* 大西洋镜像操作一遍 */
        for (int i = 0; i < m; i++) {
            accessAtl[i][n - 1] = 1;
        }
        for (int j = 0; j < n; j++) {
            accessAtl[m - 1][j] = 1;
        }

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (accessAtl[i][j] == 1) {
                    Stack<int[]> st = new Stack<>();
                    st.push(new int[]{i, j});
                    while (!st.empty()) {
                        int[] point = st.pop();
                        for (int k = 0; k < 4; k++) {
                            x = point[0] + dir[k];
                            y = point[1] + dir[k + 1];
                            if (x < 0 || x >= m || y < 0 || y >= n) continue;
                            if (heights[x][y] >= heights[point[0]][point[1]]
                                    && accessAtl[x][y] == 0) {
                                accessAtl[x][y] = 1;
                                st.push(new int[]{x, y});
                            }
                        }
                    }
                }
            }
        }
        List<List<Integer>> res = new ArrayList<>();
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (accessAtl[i][j] == 1 && accessPac[i][j] == 1) {
                    res.add(Arrays.asList(i, j));
                }
            }
        }
/*        for (int[] point : accessAtl) {
            for (int p : point) {
                System.out.print(p);
                System.out.print(" ");
            }
            System.out.println();
        }*/
        return res;

    }

    public static void main(String[] args) {
        int[][] heights = {{1,2,2,3,5},
                           {3,2,3,4,4},
                           {2,4,5,3,1},
                           {6,7,1,4,5},
                           {5,1,1,2,4}};
        System.out.println(pacificAtlantic(heights));
    }
}
```

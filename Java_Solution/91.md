91. Decode Ways (Medium)
- 题目描述
  - 已知字母 A-Z 可以表示成数字 1-26。给定一个数字串，求有多少种不同的字符串等价于这个 数字串。

```java
/*
 * Input: s = "226"
 * Output: 3
 * Explanation: "226" could be decoded as "BZ" (2 26), "VF" (22 6), or "BBF" (2 2 6).
 * 思路：if s[i - 1] == 1，s[i] == 0       -> dp[i] = dp[i - 2]
 *                        s[i] == [1,9]   -> dp[i] = dp[i - 1] + dp[i - 2]
 *                     2, s[i] == 0       -> dp[i] = dp[i - 2]
 *                        s[i] == [7,9]   -> dp[i] = dp[i - 1]
 *                        s[i] == [1,6]   -> dp[i] = dp[i - 1] + dp[i - 2]
 *               0,[3,9], s[i] == 0       -> dp[i] = 0
 *                     0, s[i] == [1,9]   -> dp[i] = dp[i - 1]
 *                 [3,9], s[i] == [1,9]   -> dp[i] = dp[i - 1]
 */

class Solution {
    public static int numDecodings(String s) {
        int n = s.length();
        int[] dp = new int[n + 1];
        char[] ch = s.toCharArray();
        Arrays.fill(dp, 1);
        int prev = ch[0] - '0';
        if (prev == 0) return 0;
        if (n == 1) return 1;
        for (int i = 2; i <= n; i++) {
            int cur = ch[i - 1] - '0';
            if (cur == 0) {
                if (prev == 1 || prev == 2) dp[i] = dp[i - 2];
                else return 0;
            } else {
                if (prev == 0 || (prev == 2 && (cur >= 7 && cur <= 9)) || (prev >= 3 && prev <= 9)) {
                    dp[i] = dp[i - 1];
                } else {
                    dp[i] = dp[i - 1] + dp[i - 2];
                }
            }
            prev = cur;
        }
        return dp[n];
    }

    public static void main(String[] args) {
        System.out.println(numDecodings(new String("1123")));
    }
}
```

### 合并两数组

```java
// 从尾部向前
class Solution {
    public void merge(int[] n1, int l1, int[] n2, int l2) {
        int tail1 = l1-1, tail2 = l2-1, fin = l1+l2-1;
        while(tail1>=0 && tail2>=0){
            n1[fin--] = n1[tail1]>n2[tail2]?n1[tail1--]:n2[tail2--];
        }
        while(tail2>=0){
            n1[fin--] = n2[tail2--];
        }
        return;
    }
}
```
